#include <bits/stdc++.h> 
vector<pair<char,int>> duplicate_char(string s, int n){
    // Write your code here.
    vector<pair<char,int>>ans;
    unordered_map<char,int>mp;
    for(int i=0; i<n; i++){
        mp[s[i]]++;
    }
    for(auto &x:mp){
        if(x.second>=2){
            ans.push_back({x.first,x.second});

        }
    }
    sort(ans.begin(),ans.end());
    return ans;
}
