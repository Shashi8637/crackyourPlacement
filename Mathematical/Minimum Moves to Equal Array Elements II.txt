class Solution {
public:
    int minMoves2(vector<int>& nums) {
        int n = nums.size();
        int low = 0;
        int high = n-1;
        sort(nums.begin(),nums.end());

        int mid = (low + high)/2;
        int ans = 0;
        for(int i=0; i<n; i++){
             ans += abs(nums[i]-nums[mid]);
        }
        return ans;
    }
};
